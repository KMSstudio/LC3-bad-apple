.ORIG x3000
    br main

PTAPE_START .fill ATAPE_START
STACK_START .fill xFE00
WHITE_BOARD .blkw xcd  

CONST_SIXTN .fill x0010
CONST_2HD56 .fill x0100

CONST_SPACE .fill x0020
CONST_FILLC .fill x0040
CONST_NEWLN .fill x000d

main
    ld  r6 STACK_START
    jsr INITL_FRAME
    ld  r4 PTAPE_START
main_loop
    jsr CSTRT_FRAME
    add r0 r0 #0
    brz main_fin
    
    jsr PRINT_FRAME
    br  main_loop
main_fin

    halt

; r0=r0>>8, r1=r0&0x00ff
DIVMOD_256
    add r6 r6 #-3
    str r2 r6 #2
    str r3 r6 #1
    str r4 r6 #0 ;set

    ld  r2 CONST_2HD56
    add r2 r2 #-1
    and r1 r0 r2 ;mod 0x0100
    
    and r3 r3 #0
    add r3 r3 #1
    add r2 r2 #1
lbl_m01
    brz lbl_m02
    and r4 r0 r3
    brz lbl_m03
    add r0 r0 r2
    lbl_m03
    add r2 r2 r2
    add r3 r3 r3
    br  lbl_m01
lbl_m02
    
    ld  r2 CONST_2HD56
    add r2 r2 #-1
    and r0 r0 r2 ;div 0x0100
    
    ldr r2 r6 #2
    ldr r3 r6 #1
    ldr r4 r6 #0
    add r6 r6 #3 ;load
    ret

; r0=r0>>4, r1=r0&0x0f
DIVMOD_16
    add r6 r6 #-3
    str r2 r6 #2
    str r3 r6 #1
    str r4 r6 #0 ;set

    ld  r2 CONST_SIXTN
    add r2 r2 #-1
    and r1 r0 r2 ;mod 0x10
    
    and r3 r3 #0
    add r3 r3 #1
    add r2 r2 #1
lbl_mc01
    brz lbl_mc02
    and r4 r0 r3
    brz lbl_mc03
    add r0 r0 r2
    lbl_mc03
    add r2 r2 r2
    add r3 r3 r3
    br  lbl_mc01
lbl_mc02
    
    ld  r2 CONST_SIXTN
    add r2 r2 #-1
    and r0 r0 r2 ;div 0x10
    
    ldr r2 r6 #2
    ldr r3 r6 #1
    ldr r4 r6 #0
    add r6 r6 #3 ;load
    ret

; r0=r0*0x10 (16)
MUL_16
    add r0 r0 r0
    add r0 r0 r0
    add r0 r0 r0
    add r0 r0 r0
    ret

; pix[r0] = !pix[r0]
XOR_PIX
    add r6 r6 #-2
    str r1 r6 #1
    str r2 r6 #0

    ld  r1 PHITE_BOARD
    add r1 r1 r0 ;r1=loc
    
    ldr r0 r1 #0
    not r0 r0
    add r0 r0 #1
    
    ld  r2 CONST_SPACE
    add r0 r0 r2
    
    brnp lbl_x01
    ld  r2 CONST_FILLC
    lbl_x01
    
    str r2 r1 #0
    
    ldr r1 r6 #1
    ldr r2 r6 #0
    add r6 r6 #2
    ret

ATAPE_FLAG  .fill x0
;r0 = (mem[r4]>>8) if FLAG else (mem[r4]&0xff)
LOAD_ATAPE
    add r6 r6 #-1
    str r7 r6 #0
    add r6 r6 #-2
    str r1 r6 #1
    str r2 r6 #0

    ldr r0 r4 #0
    jsr DIVMOD_256
    
    ld  r2 ATAPE_FLAG
    brz lbl_la01
    add r0 r1 #0
    and r2 r2 #0
    add r4 r4 #1    ;flag=1
    br lbl_la02
    lbl_la01
    add r2 r2 #1    ;flog=0
    
    lbl_la02
    st  r2 ATAPE_FLAG
    
    ldr r1 r6 #1
    ldr r2 r6 #0
    add r6 r6 #2
    ldr r7 r6 #0
    add r6 r6 #1
    
    ret
    
;initalize_frame
INITL_FRAME
    add r6 r6 #-5
    str r1 r6 #4
    str r2 r6 #3
    str r3 r6 #2
    str r4 r6 #1
    str r5 r6 #0

    ld  r4 PHITE_BOARD
    ld  r5 CONST_SPACE
    ld  r3 CONST_NEWLN

    and r1 r1 #0
    add r1 r1 x0c
lbl_i01
    brz lbl_i03
    and r2 r2 #0
    ld  r2 CONST_SIXTN
    lbl_i02
        brz lbl_i04
        str r5 r4 #0
        add r4 r4 #1
        add r2 r2 #-1
        br  lbl_i02
    lbl_i04
    str r3 r4 #0
    add r4 r4 #1
    add r1 r1 #-1
    br lbl_i01
lbl_i03
    and r5 r5 #0
    str r5 r4 #0

    ldr r1 r6 #4
    ldr r2 r6 #3
    ldr r3 r6 #2
    ldr r4 r6 #1
    ldr r5 r6 #0
    add r6 r6 #5
    
    ret

;construct_frame
CSTRT_FRAME
    add r6 r6 #-1
    str r7 r6 #0
    
    jsr LOAD_ATAPE
    add r5 r0 #0
    
    ld  r3 CONST_2HD56
    not r3 r3
    add r3 r3 #2
    add r0 r0 r3
    
    brn lbl_cf03
    br  lbl_cffn
    lbl_cf03
    
    ld  r3 CONST_SIXTN
    not r3 r3
    add r3 r3 #1
    add r3 r3 r5
    brzp lbl_cf01
    not r3 r3
    add r3 r3 #1
lbl_cf02   
    brz lbl_cf01
    ;;;;;;;;;;;;;;;;;;;;;;;;;;;;;;
    add r3 r3 #-1
    br  lbl_cf02
lbl_cf01 ;skip delay

    add r5 r5 #0
lbl_cfloop
    brz lbl_cffn
    
    jsr LOAD_ATAPE ;row-col information
    jsr DIVMOD_16
    add r2 r0 #0
    jsr MUL_16
    add r0 r0 r2 ;r0 *= 0x11
    
    add r0 r0 r1 ;loc=r0+r1
    jsr XOR_PIX
    
    add r5 r5 #-1
    br  lbl_cfloop
lbl_cffn

    and r0 r0 #0
    add r5 r5 #0
    brp lcl_cf11
    add r0 r0 #1
    lcl_cf11

    ldr r7 r6 #0
    add r6 r6 #1
    
    ret

PONST_SPACE .fill CONST_SPACE
PONST_FILLC .fill CONST_FILLC
PONST_NEWLN .fill CONST_NEWLN
PHITE_BOARD .fill WHITE_BOARD
    
PRINT_FRAME
    ld  r0 PHITE_BOARD
    puts
    ret

.end
